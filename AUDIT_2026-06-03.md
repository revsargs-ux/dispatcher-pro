### 1.1 `routes/payment-routes.js` — Двойное объявление `session` (ReferenceError → краш)
**Файл:** `routes/payment-routes.js`, строки 15 и 30  
**Описание:** В `handleExportCsv` переменная `const session` объявляется дважды — на строке 15 (для проверки роли) и на строке 30 (для аудита). Это вызовет `SyntaxError: Identifier 'session' has already been declared` → функция крашится, CSV экспорт **не работает вообще**.

```js
// Баг:
async function handleExportCsv(req, res, cors) {
  const session = requireAuth(req);  // строка 15
  // ...
```

```js
async function handleExportCsv(req, res, cors) {
  const session = requireAuth(req);

    return json(res, { error: 'Нет доступа' }, 403, cors);
  }
  try {
    // ... весь код ...
    audit('export', 'payments.csv', session.userId, session.role, ...);
    // Убрать второе const session
```

### 1.3 `routes/user-routes.js` — handleClientPayMethodGet: нет авторизации
**Файл:** `routes/user-routes.js`, функция `handleClientPayMethodGet`  
**Описание:** Функция не проверяет `requireAuth` внутренне. Маршрут в `routes.js` проверяет `auth()`, но нет проверки что запрашивающий имеет право видеть данные именно этого `client_id`. Любой авторизованный пользователь может узнать способ оплаты любого клиента.

**Исправление:**
```js
function handleClientPayMethodGet(req, res, cors) {
  const session = requireAuth(req);
  const cid = new URL('http://localhost' + req.url).searchParams.get('client_id');
  if (!session) return json(res, { error: 'Auth required' }, 401, cors);
  if (session.role !== 'owner' && session.role !== 'dispatcher' && session.userId !== cid) {
    return json(res, { error: 'Нет доступа' }, 403, cors);
  }
  // ...
}
```

### 1.4 `modules/telegram.js` — SQL injection через phone в ilike запросах
**Файл:** `modules/telegram.js`, функции `tgNotify`, `linkTgUser`  
**Описание:** Номер телефона подставляется напрямую в Supabase query без санитизации. Хотя Supabase PostgREST не позволяет классический SQL injection, спецсимволы `%` и `_` в ilike паттернах могут привести к неожиданным совпадениям. Аналогичная проблема в `max-bot.js`.

```js
// Текущий код:
`phone=ilike.%25${cleanPhone.slice(-10)}%25`
// Если cleanPhone содержит спецсимволы — непредсказуемое поведение
```

**Исправление:**
```js
// Санитизировать для ilike:
const safe = cleanPhone.slice(-10).replace(/[%_]/g, '');
`phone=ilike.%25${safe}%25`
```

### 1.5 `modules/telegram.js` + `modules/max-bot.js` — tryForwardChat: SQL injection через shift IDs
**Файл:** `modules/telegram.js` строка ~180, `modules/max-bot.js` строка ~165  
**Описание:** `shiftIds.join(',')` подставляется в PostgREST `id=in.(...)` без валидации. Если `shift_id` из БД окажется скомпрометирован, возможна инъекция в PostgREST запрос.

**Исправление:**
```js
const validIds = shiftIds.filter(id => /^[0-9a-f-]{36}$/.test(id));
if (!validIds.length) return false;
const activeRes = await fetch(`...id=in.(${validIds.join(',')})...`);
```

### 1.6 `server.js` — Shift photos: path traversal
**Файл:** `server.js`, endpoint `/shift-photos/`  
**Описание:** Имя файла очищается через `replace(/[^a-zA-Z0-9._-]/g, '_')`, но при извлечении файла из URL `urlPath.replace('/shift-photos/', '')` не проверяется отсутствие `..`. Защита `!photoFile.startsWith(shiftPhotosDir)` есть, но URL не декодируется — возможен обход через URL encoding (`%2e%2e`).

**Исправление:**
```js
const rawName = decodeURIComponent(urlPath.replace('/shift-photos/', ''));
const photoFile = path.join(shiftPhotosDir, rawName);
// Добавить проверку на ..
if (rawName.includes('..') || !photoFile.startsWith(shiftPhotosDir)) {
  res.writeHead(403); return res.end('Forbidden');
}
```
